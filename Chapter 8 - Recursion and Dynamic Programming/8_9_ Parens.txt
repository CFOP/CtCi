void parens(int n, int close = 0, string s = ""){
	if(!n){
		std::cout<<s;
		while(close--)
			std::cout<<')';
		return;
	}

	if(n){
		parens(n-1, close+1, s+'(');
	}

	if(close){
		parens(n, close-1, s+')');
	}

}
