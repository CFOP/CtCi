uint32_t insertion(uint32_t N, uint32_t M, uint8_t i, uint8_t j){
	uint32_t mask = 1<<j;
	mask -= 1;
	mask &= ~(1<<i)-1;
	mask = ~mask;

	N &= mask;

	return N | M<<(i);
}
